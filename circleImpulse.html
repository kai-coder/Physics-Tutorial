def dist(p1, p2):
    # Difference between circle coords
    dx = p1.x - p2.x
    dy = p1.y - p2.y

    # Calculate distance with Pythagorean Theorem
    return sqrt(dx * dx + dy * dy)


def circleCircle(c1, c2):
    # Return the truth value of
    # distance <= sum of radii
    return dist(c1, c2) <= c1.r + c2.r
